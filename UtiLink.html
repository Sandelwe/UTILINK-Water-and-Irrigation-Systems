            if (pipesTable) {

                pipesTable.classList.add('hidden');

            }

 

            // Update active link styling (optional)

            const navLinks = document.querySelectorAll('.nav-link');

            navLinks.forEach(link => {

                link.classList.remove('font-bold', 'border-b-2', 'border-utilink-red');

                if (link.onclick.toString().includes(pageId)) {

                    link.classList.add('font-bold', 'border-b-2', 'border-utilink-red');

                }

            });

        }
